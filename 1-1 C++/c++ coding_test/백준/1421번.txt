#include<iostream>
#include<algorithm>
using namespace std;


int main() {
	int n, c, w;

	int max=0;
	cin >> n >> c >> w;
	int* arr = new int[1001];
	int oak = 0;
	int cut_attempt = 0;
	int result = 0;
	int tmp_result = 0;

	for (int i = 0; i < n; i++) {
		cin >> arr[i];
		if (max < arr[i])
			max = arr[i];
	}

	
	
	for (int i = 1; i <=max;i++) {
		oak = 0;
		cut_attempt = 0;
		
			for (int j = 0; j < n; j++) {
				if (i <= arr[j]) {
					if (arr[j] % i == 0) { //나눠떨어질때
						oak += int(arr[j] / i);
						cut_attempt += int(arr[j] / i) - 1;
					}
					else if (arr[j] % i != 0) {
						oak += int(arr[j] / i);
						cut_attempt += int(arr[j] / i);
					}
				}
			}

			tmp_result = oak * w * i - c * cut_attempt;
			
			if (tmp_result > result)
				result = tmp_result;
		

	}
	cout << result;
}
