#include<iostream>
#include<string>
using namespace std;

int main() {
	int n;
	cin >> n;

	char student[101] = { NULL, };

	string arr[60];

	for (int i = 1; i <= n; i++) {
		string temp;
		getline(cin, temp);
		arr[i] = temp;
	}

	for (int i = 0; i < 2*n - 1; i++) {
		int temp_num;
		char temp_ch;
		cin >> temp_num >> temp_ch;

		if (student[temp_num]=='A') {
			if (temp_ch == 'B')
				student[temp_num] = NULL;
		}

		else if (student[temp_num]=='B') {
			if (temp_ch == 'A')
				student[temp_num] = NULL;
		}
		else
			student[temp_num] = temp_ch;
	}

	for (int i = 1; i <= n; i++) {
		if (student[i] != NULL)
			cout << i<<arr[i];
	}

}
